> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Complete the Expo mobile app (Teacher + Parent/Student experiences), wire up Expo Push Notifications, build the report card PDF via Supabase Edge Function, and deploy web to Vercel + mobile APK via EAS Build.

**Architecture:** Mobile mirrors the web teacher portal. Push tokens stored in Supabase and sent via Expo Push API from Edge Functions triggered on DB events. Report card PDF is generated by a Deno Edge Function that fetches student data, injects it into the HTML template, and returns a PDF via `html-pdf-chrome` equivalent in Deno. Vercel deployment is zero-config from the `apps/web` directory. EAS Build produces an APK.

**Tech Stack:** Expo SDK 52, Expo Router, NativeWind 4, expo-notifications, Supabase Edge Functions (Deno), Vercel CLI, EAS CLI

```tsx
import { useEffect, useState } from "react";
import { View, Text, ScrollView, ActivityIndicator } from "react-native";
import { supabase } from "../../lib/supabase";

interface TimetableSlot {
  period: number;
  subject: string;
  section: string;
}

export default function TeacherDashboard() {
  const [slots, setSlots] = useState<TimetableSlot[]>([]);
  const [loading, setLoading] = useState(true);
  const [teacherName, setTeacherName] = useState("");

  const DAYS = ["", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"];
  const today = new Date().getDay() || 7;

  useEffect(() => {
    async function load() {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return;

      const { data: profile } = await supabase
        .from("profiles")
        .select("school_id, full_name")
        .eq("id", user.id)
        .single();

      setTeacherName(profile?.full_name ?? "");

      const { data: schedule } = await supabase
        .from("timetable")
        .select("period, subject:subjects(name), section:sections(name)")
        .eq("school_id", profile?.school_id)
        .eq("teacher_id", user.id)
        .eq("day_of_week", today)
        .order("period");

      setSlots(
        (schedule ?? []).map((s) => ({
          period: s.period,
          subject: (s.subject as { name: string } | null)?.name ?? "",
          section: (s.section as { name: string } | null)?.name ?? "",
        }))
      );
      setLoading(false);
    }
    load();
  }, [today]);

  if (loading) {
    return <ActivityIndicator className="flex-1 mt-20" />;
  }

  return (
    <ScrollView className="flex-1 bg-gray-50 p-5">
      <Text className="mt-12 text-2xl font-bold text-gray-900">
        Hi, {teacherName}
      </Text>
      <Text className="mb-5 text-sm text-gray-500">{DAYS[today]}'s classes</Text>

      {slots.length === 0 ? (
        <Text className="text-sm text-gray-400">No classes today.</Text>
      ) : (
        slots.map((s) => (
          <View key={s.period} className="mb-2 flex-row items-center rounded-xl bg-white p-4 shadow-sm">
            <Text className="mr-4 w-8 text-center text-sm text-gray-400">P{s.period}</Text>
            <Text className="flex-1 font-medium text-gray-800">{s.subject}</Text>
            <Text className="text-sm text-gray-500">Sec {s.section}</Text>
          </View>
        ))
      )}
    </ScrollView>
  );
}
```

- [ ] **Step 4: Create `apps/mobile/app/(teacher)/attendance.tsx`**

```tsx
import { useEffect, useState } from "react";
import {
  View, Text, ScrollView, TouchableOpacity, ActivityIndicator, Alert
} from "react-native";
import { supabase } from "../../lib/supabase";

type AttendanceStatus = "present" | "absent" | "late" | "half_day";
interface Section { id: string; name: string; className: string }
interface Student { profileId: string; name: string }

const STATUS_COLORS: Record<AttendanceStatus, string> = {
  present: "bg-green-100",
  absent: "bg-red-100",
  late: "bg-yellow-100",
  half_day: "bg-orange-100",
};

export default function TeacherAttendance() {
  const [sections, setSections] = useState<Section[]>([]);
  const [selectedSection, setSelectedSection] = useState<Section | null>(null);
  const [students, setStudents] = useState<Student[]>([]);
  const [statuses, setStatuses] = useState<Record<string, AttendanceStatus>>({});
  const [schoolId, setSchoolId] = useState("");
  const [userId, setUserId] = useState("");
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);

  const today = new Date().toISOString().split("T")[0];

  useEffect(() => {
    async function load() {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return;
      setUserId(user.id);

      const { data: profile } = await supabase
        .from("profiles").select("school_id").eq("id", user.id).single();
      setSchoolId(profile?.school_id ?? "");

      const { data: tt } = await supabase
        .from("timetable")
        .select("section_id, section:sections(id, name, class:classes(name))")
        .eq("school_id", profile?.school_id)
        .eq("teacher_id", user.id);

      const unique = Array.from(
        new Map((tt ?? []).map((s) => [
          s.section_id,
          {
            id: s.section_id,
            name: (s.section as { name: string; class: { name: string } } | null)?.name ?? "",
            className: (s.section as { name: string; class: { name: string } } | null)?.class?.name ?? "",
          },
        ])).values()
      );
      setSections(unique);
      setLoading(false);
    }
    load();
  }, []);

  async function selectSection(section: Section) {
    setSelectedSection(section);
    const { data: studentData } = await supabase
      .from("student_profiles")
      .select("profile_id, profile:profiles(full_name)")
      .eq("section_id", section.id)
      .eq("school_id", schoolId);

    const studentList = (studentData ?? []).map((s) => ({
      profileId: s.profile_id,
      name: (s.profile as { full_name: string } | null)?.full_name ?? "",
    }));
    setStudents(studentList);
    setStatuses(Object.fromEntries(studentList.map((s) => [s.profileId, "present" as AttendanceStatus])));
  }

  async function saveAttendance() {
    setSaving(true);
    const records = students.map((s) => ({
      school_id: schoolId,
      student_id: s.profileId,
      section_id: selectedSection!.id,
      date: today,
      status: statuses[s.profileId],
      marked_by: userId,
    }));

    try {
      await supabase.from("attendance_records").upsert(records, { onConflict: "student_id,date" });
      setSaving(false);
      Alert.alert("Saved", "Attendance saved successfully.");
      setSelectedSection(null);
    } catch {
      // Offline: save to AsyncStorage for later sync
      const AsyncStorage = require("@react-native-async-storage/async-storage").default;
      const pending = JSON.parse(await AsyncStorage.getItem("pending_attendance") ?? "[]");
      pending.push({ records, timestamp: Date.now() });
      await AsyncStorage.setItem("pending_attendance", JSON.stringify(pending));
      setSaving(false);
      Alert.alert("Saved Offline", "Attendance saved locally. It will sync when you're back online.");
      setSelectedSection(null);
    }
  }

  if (loading) return <ActivityIndicator className="flex-1 mt-20" />;

  return (
    <ScrollView className="flex-1 bg-gray-50 p-5">
      <Text className="mt-12 mb-5 text-2xl font-bold text-gray-900">Attendance</Text>
      <Text className="mb-3 text-sm text-gray-500">{today}</Text>

      {!selectedSection ? (
        <>
          <Text className="mb-3 text-sm font-medium text-gray-600">Select Section</Text>
          {sections.map((s) => (
            <TouchableOpacity
              key={s.id}
              onPress={() => selectSection(s)}
              className="mb-2 rounded-xl bg-white p-4 shadow-sm"
            >
              <Text className="font-medium text-gray-800">{s.className} — {s.name}</Text>
            </TouchableOpacity>
          ))}
        </>
      ) : (
        <>
          <Text className="mb-1 font-medium text-gray-700">
            {selectedSection.className} — Sec {selectedSection.name}
          </Text>
          <View className="mb-4 flex-row gap-2">
            {(["present", "absent"] as AttendanceStatus[]).map((status) => (
              <TouchableOpacity
                key={status}
                onPress={() =>
                  setStatuses(Object.fromEntries(students.map((s) => [s.profileId, status])))
                }
                className="rounded-lg bg-white px-3 py-2 shadow-sm"
              >
                <Text className="text-sm capitalize text-gray-700">All {status}</Text>
              </TouchableOpacity>
            ))}
          </View>
          {students.map((s) => (
            <View key={s.profileId} className="mb-2 rounded-xl bg-white p-3 shadow-sm">
              <Text className="mb-2 font-medium text-gray-800">{s.name}</Text>
              <View className="flex-row flex-wrap gap-1">
                {(["present", "absent", "late", "half_day"] as AttendanceStatus[]).map((opt) => (
                  <TouchableOpacity
                    key={opt}
                    onPress={() => setStatuses((prev) => ({ ...prev, [s.profileId]: opt }))}
                    className={`rounded px-2 py-1 ${statuses[s.profileId] === opt ? STATUS_COLORS[opt] : "bg-gray-100"}`}
                  >
                    <Text className="text-xs capitalize text-gray-700">
                      {opt === "half_day" ? "Half" : opt}
                    </Text>
                  </TouchableOpacity>
                ))}
              </View>
            </View>
          ))}
          <TouchableOpacity
            onPress={saveAttendance}
            disabled={saving}
            className="mt-4 rounded-xl bg-blue-600 py-4"
          >
            <Text className="text-center font-medium text-white">
              {saving ? "Saving…" : "Save Attendance"}
            </Text>
          </TouchableOpacity>
        </>
      )}
    </ScrollView>
  );
}
```

```typescript
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";

interface PushPayload {
  to: string;
  title: string;
  body: string;
  data?: Record<string, unknown>;
}

serve(async (req) => {
  const { pushToken, title, body, data } = await req.json() as {
    pushToken: string;
    title: string;
    body: string;
    data?: Record<string, unknown>;
  };

  const message: PushPayload = { to: pushToken, title, body, data };

  const response = await fetch("https://exp.host/--/api/v2/push/send", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Accept: "application/json",
    },
    body: JSON.stringify(message),
  });

  const result = await response.json();
  return new Response(JSON.stringify(result), {
    headers: { "Content-Type": "application/json" },
  });
});
```

```typescript
  const { studentId, examId } = await req.json() as { studentId: string; examId: string };

  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  );

  // Fetch student profile
  const { data: student } = await supabase
    .from("profiles")
    .select("full_name, email")
    .eq("id", studentId)
    .single();

  // Fetch student class/section
  const { data: sp } = await supabase
    .from("student_profiles")
    .select("roll_number, class:classes(name), section:sections(name)")
    .eq("profile_id", studentId)
    .single();

  // Fetch exam details
  const { data: exam } = await supabase
    .from("exams")
    .select("name, school_id")
    .eq("id", examId)
    .single();

  // Fetch results for this student + exam
  const { data: results } = await supabase
    .from("exam_results")
    .select("marks_obtained, max_marks, grade, subject:subjects(name)")
    .eq("student_id", studentId)
    .eq("exam_id", examId);

  // Fetch report card template
  const { data: template } = await supabase
    .from("report_card_templates")
    .select("html_template")
    .eq("school_id", exam?.school_id)
    .eq("is_default", true)
    .single();

  const htmlTemplate = template?.html_template ?? DEFAULT_TEMPLATE;

  // Calculate totals
  const totalMarks = (results ?? []).reduce((sum, r) => sum + (r.marks_obtained ?? 0), 0);
  const maxTotal = (results ?? []).reduce((sum, r) => sum + r.max_marks, 0);
  const percentage = maxTotal > 0 ? ((totalMarks / maxTotal) * 100).toFixed(1) : "0";

  // Build results rows HTML
  const resultsRows = (results ?? [])
    .map(
      (r) =>
        `<tr>
          <td>${(r.subject as { name: string } | null)?.name ?? ""}</td>
          <td>${r.marks_obtained ?? "—"}</td>
          <td>${r.max_marks}</td>
          <td>${r.grade ?? "—"}</td>
        </tr>`
    )
    .join("");

  // Inject data into template
  const html = htmlTemplate
    .replace("{{student_name}}", student?.full_name ?? "")
    .replace("{{roll_number}}", sp?.roll_number ?? "")
    .replace("{{class}}", (sp?.class as { name: string } | null)?.name ?? "")
    .replace("{{section}}", (sp?.section as { name: string } | null)?.name ?? "")
    .replace("{{exam_name}}", exam?.name ?? "")
    .replace("{{results_rows}}", resultsRows)
    .replace("{{total_marks}}", `${totalMarks}`)
    .replace("{{max_marks}}", `${maxTotal}`)
    .replace("{{percentage}}", `${percentage}%`);

  return new Response(html, {
    headers: {
      "Content-Type": "text/html",
    },
  });
});

const DEFAULT_TEMPLATE = `<!DOCTYPE html>
<html>
<head>
  <style>
    body { font-family: Arial, sans-serif; padding: 32px; }
    h1 { font-size: 20px; margin-bottom: 4px; }
    .meta { font-size: 13px; color: #555; margin-bottom: 24px; }
    table { width: 100%; border-collapse: collapse; }
    th, td { border: 1px solid #ddd; padding: 8px 12px; text-align: left; }
    th { background: #f5f5f5; }
    .footer { margin-top: 24px; font-size: 13px; }
  </style>
</head>
<body>
  <h1>Report Card — {{exam_name}}</h1>
  <div class="meta">
    <strong>{{student_name}}</strong> · Roll No: {{roll_number}} · Class {{class}} - Sec {{section}}
  </div>
  <table>
    <thead>
      <tr><th>Subject</th><th>Marks Obtained</th><th>Max Marks</th><th>Grade</th></tr>
    </thead>
    <tbody>{{results_rows}}</tbody>
  </table>
  <div class="footer">
    <strong>Total: {{total_marks}} / {{max_marks}} ({{percentage}})</strong>
  </div>
</body>
</html>`;
```

The actual school name and bundle ID are overridden per-build via the EAS profile's `env` block and `app.config.js`. Create `apps/mobile/app.config.js`:

```javascript
  return {
    ...config,
    name: process.env.EXPO_PUBLIC_SCHOOL_NAME ?? config.name,

      ? process.env.EXPO_PUBLIC_BUNDLE_ID.split(".").pop()
      : config.slug,
    android: {
      ...config.android,
      package: process.env.EXPO_PUBLIC_BUNDLE_ID ?? config.android?.package,
    },
    ios: {
      ...config.ios,
      bundleIdentifier: process.env.EXPO_PUBLIC_BUNDLE_ID ?? config.ios?.bundleIdentifier,
    },
  };
};
```
